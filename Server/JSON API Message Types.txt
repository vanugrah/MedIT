Message Types

Logging In
  Request
    MessageType : String = "AuthRequest"
    Username : String
    Password : String
  Response
    MessageType : String = "AuthResponse"
    Success : Boolean
    User : User

Looking up a patient
  Request
    MessageType : String = "PatientQuery"
    PatientID : int
  Response
    MessageType : String = "PatientQueryResults"
    Found : Boolean
    Patient : Patient
    
Looking up appointments
  Request
    MessageType : String = "AppointmentsQuery"
    Username : String
  Response
    MessageType : String = "AppointmentsQueryResults"
    Appointments : Appointment[]

Looking up a single appointment
  Request
    MessageType : String = "AppointmentQuery"
    AppointmentID : int
  Response
    Appointment : Appointment
    
Create a new user
  Request
    MessageType : String = "UserCreate"
    Username : String
    Password : String
    PhoneNumber : String
    EmailAddress : String
    FirstName : String
    LastName : String
  Response
    MessageType : String = "UserCreateConfirmation"
    User : User
    
Create a new patient
  Request
    MessageType : String = "PatientCreate"
  Response
    MessageType : String = "PatientCreateConfirmation"
    Patient : Patient
    
Save user settings
  Request
    MessageType : String = "UserSettingsChange"
    Username : String
    GetsPush : Boolean
    GetsSMS : Boolean
    GetsEmail : Boolean
  Response
    MessageType : String = "UserSettingsChangeConfirmation"

Confirming an appointment
  Request
    MessageType : String = "ConfirmAppointment"
    AppointmentID : int
  Response
    MessageType : String = "AppointmentConfirmed"
    AppointmentID : int
  
Cancel an appointment
  Request
    MessageType : String = "CancelAppointment"
    AppointmentID : int
  Response
    MessageType : String = "AppointmentCancelled"
    AppointmentID : int
    
Check-In for an appointment
  Request
    MessageType : String = "CheckInForAppointment"
    AppointmentID : int
  Response
    MessageType : String = "AppointmentCheckedIn"
    AppointmentID : int
